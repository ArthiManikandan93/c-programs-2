#include<stdio.h>
#include<string.h>
int main() {
    char str[100];
    int i=0,uppercase=0;
    scanf("%s",str);
  
    while (str[i] != '\0') {
        if(str[i] >='A' && str[i]<='Z'){
         uppercase=1;
         break;
        }
           i++;
    }
    if(uppercase=1){
        printf("Vaild");
    }
    else{
        printf("Not Valid");
    }
    return 0;
}
