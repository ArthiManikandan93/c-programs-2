#include <stdio.h>

int main() {
    int capacity;
    int N;
    int i = 0;
    int load;
    int safeHours = 0;
    int failureCount = 0;

    scanf("%d", &capacity);
    scanf("%d", &N);

    while (i < N) {
        scanf("%d", &load);

        if (load > capacity) {
            failureCount++;
        } else {
            safeHours++;
        }

        i++;
    }

    printf("Safe Hours: %d\n", safeHours);
    printf("Failure Count: %d", failureCount);

    return 0;
}
