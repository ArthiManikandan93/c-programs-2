#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int sum = 0;
    int roll;

    for(int i = 0; i < N-1; i++) {
        scanf("%d", &roll);
        sum += roll;
    }

    int expectedSum = N * (N + 1) / 2;
    int missingRoll = expectedSum - sum;

    printf("%d\n", missingRoll);

    return 0;
}
