#include <stdio.h>
#include <math.h>

int main() {
    int N;
    scanf("%d", &N);

    int original = N;
    int temp = N;
    int digits = 0;
    int sum = 0;

    // Count number of digits
    for(; temp > 0; temp = temp / 10) {
        digits++;
    }

    temp = N;  // Reset temp

    // Calculate Armstrong sum
    for(; temp > 0; temp = temp / 10) {
        int digit = temp % 10;
        sum += pow(digit, digits);
    }

    if(sum == original) {
        printf("Yes\n");
    } else {
        printf("No\n");
    }

    return 0;
}
