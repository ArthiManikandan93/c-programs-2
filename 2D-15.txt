#include <stdio.h>

int main() {
    int R,C;
    scanf("%d %d",&R,&C);
    int a[R][C];
    for (int i=0;i<R;i++){
        for(int j=0;j<C;j++){
            scanf("%d",&a[i][j]);
        }
    }
    int count=0;
    
    for(int i=0;i<R;i++){
    for(int j=0;j<C;j++){
        int n=a[i][j];
          int prime=1;
          if(n<=1){
              prime=0;
          }
          for(int k=2;k<n;k++){
             if(n%k==0){
                 prime=0;
                 break;
             }
          }
        if(prime==1){
         count++;
        }
    } 
 }
   
    printf("%d",count);

    return 0;
}
