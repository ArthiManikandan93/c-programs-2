#include <stdio.h>
#include <string.h>
int main() {
    char str[100];
    int count=0,i=0;
    scanf("%s",str);
   
    while(str[i]!='\0'){
        if(str[i] =='a'||str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u'||str[i]=='A'||str[i]=='E'||str[i]=='U'||str[i]=='O'||str[i]=='I'){
            count++;
        }
        i++;
    }
   printf("%d",count);
   

    return 0;
}
