#include <stdio.h>

int main() {
    int r, c;
    scanf("%d %d", &r, &c);

    int a[r][c];

    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    int found = 0;

    for(int i = 0; i < r && !found; i++) {
        for(int j = 0; j < c && !found; j++) {

            for(int x = i; x < r; x++) {
                for(int y = (x == i ? j + 1 : 0); y < c; y++) {

                    if(a[i][j] == a[x][y]) {
                        printf("%d", a[i][j]);
                        found = 1;
                        break;
                    }
                }
                if(found) break;
            }
        }
    }

    return 0;
}
