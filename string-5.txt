
#include <stdio.h>
#include <string.h>
int main() {
    char str[100];
    int len,i;
    int flag=1;
    scanf("%s",str);
    len=strlen(str);
    for(int i=0;i<len/2;i++){
        if(str[i] != str[len-i-1]){
            flag=0;
        }
    }
    if(flag==1){
        printf("Palindrome");
    }
    else{
        printf("Not a Palindrome");
    }
   

    return 0;
}
