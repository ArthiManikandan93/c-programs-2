#include <stdio.h>

int main() {
    int N, K;
    scanf("%d", &N);
    int arr[N];
    for(int i = 0; i < N; i++) {
        scanf("%d", &arr[i]);
    }


    scanf("%d", &K);

    
    for(int i = K; i < N; i++) {
        printf("%d ", arr[i]);
    }

    for(int i = 0; i < K; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}
