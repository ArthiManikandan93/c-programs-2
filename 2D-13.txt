
#include <stdio.h>

int main() {
    int R,C;
    scanf("%d %d",&R,&C);
    int a[R][C];
    for (int i=0;i<R;i++){
        for(int j=0;j<C;j++){
            scanf("%d",&a[i][j]);
        }
    }
    int count=0;
   
    for(int i=0;i<R;i++){
          int start=0;
          int end=C-1;
          int palindrome=1;
        while(start<end){
            if(a[i][start]!=a[i][end]){
               palindrome =0;
                break;}
                start++;
                end--;
            
        }
        if(palindrome==1){
           count++;
        }
       }
   
    printf("%d",count);

    return 0;
}
