#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    int left = 0;
    int right = n - 1;

    while(left < right) {
        if(arr[left] != arr[right]) {
            printf("No");
            return 0;
        }
        left++;
        right--;
    }

    printf("Yes");
    return 0;
}
