#include <stdio.h>

int main() {
    int maxWeight;
    int N;
    int i = 0;
    int weight;
    int totalWeight = 0;
    int peopleEntered = 0;
    int overload = 0;

    scanf("%d", &maxWeight);
    scanf("%d", &N);

    while (i < N) {
        scanf("%d", &weight);

        if (totalWeight + weight <= maxWeight) {
            totalWeight += weight;
            peopleEntered++;
        } else {
            overload = 1;
            break;   // Stop when overload occurs
        }

        i++;
    }

    printf("People Entered: %d\n", peopleEntered);

    if (overload == 1) {
        printf("Overload Status: Yes");
    } else {
        printf("Overload Status: No");
    }

    return 0;
}
