#include <stdio.h>
#include <stdlib.h>
int main() {
    int R, C;
    scanf("%d %d", &R, &C);

    int a[R][C];

    for(int i = 0; i < R; i++){
        for(int j = 0; j < C; j++){
            scanf("%d", &a[i][j]);
        }
    }

    int maxdiff = 0;

    for(int i = 0; i < R; i++){
      for(int j = 0; j < C; j++){
          for(int k=i+1;k<R;k++){
              for(int l=0;l<C;l++){
                  int diff=abs(a[i][j]-a[k][l]);
                  if(diff>maxdiff){
                      maxdiff=diff;
                  }
              }
           }
        }
    }

    printf("%d", maxdiff);

    return 0;
}
