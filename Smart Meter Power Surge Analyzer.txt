#include <stdio.h>

int main() {
    int N;
    int i = 0;
    int usage;
    int maxUsage;
    int surgeHours = 0;

    scanf("%d", &N);

    // Read first value separately to initialize maxUsage
    scanf("%d", &usage);
    maxUsage = usage;

    if (usage > 5) {
        surgeHours++;
    }

    i = 1;

    while (i < N) {
        scanf("%d", &usage);

        if (usage > maxUsage) {
            maxUsage = usage;
        }

        if (usage > 5) {
            surgeHours++;
        }

        i++;
    }

    printf("Max Usage: %d\n", maxUsage);
    printf("Surge Hours: %d", surgeHours);

    return 0;
}
