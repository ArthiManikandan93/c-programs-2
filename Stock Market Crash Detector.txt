#include <stdio.h>

int main() {
    int N;
    int i = 1;
    int crashDay = -1;
    int totalDrops = 0;
    int consecutiveDrops = 0;

    scanf("%d", &N);

    int index[N];

    for (int j = 0; j < N; j++) {
        scanf("%d", &index[j]);
    }

    while (i < N) {
        if (index[i] < index[i - 1]) {
            totalDrops++;
            consecutiveDrops++;

            if (consecutiveDrops == 3 && crashDay == -1) {
                crashDay = i + 1;
            }
        } else {
            consecutiveDrops = 0;
        }

        i++;
    }

    if (crashDay == -1)
        printf("Crash Day: Not Detected\n");
    else
        printf("Crash Day: %d\n", crashDay);

    printf("Total Drops: %d", totalDrops);

    return 0;
}
