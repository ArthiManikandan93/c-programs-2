#include <stdio.h>

int main() {
    int fuel;
    int N;
    int i = 0;
    int burn;
    int emergencyStage = -1;
    int fuelWasted = 0;

    scanf("%d", &fuel);
    scanf("%d", &N);

    while (i < N) {
        scanf("%d", &burn);

        fuel -= burn;

        if (fuel < 0 && emergencyStage == -1) {
            emergencyStage = i + 1;      // Stage number (1-based)
            fuelWasted = -fuel;          // Extra fuel required
        }

        i++;
    }

    if (emergencyStage == -1)
        printf("Emergency Stage: Not Occurred\n");
    else
        printf("Emergency Stage: %d\n", emergencyStage);

    printf("Fuel Wasted: %d", fuelWasted);

    return 0;
}
