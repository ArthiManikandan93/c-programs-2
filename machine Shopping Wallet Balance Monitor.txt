#include <stdio.h>

int main() {
    int walletBalance;
    int N;
    int i = 0;
    int purchase;
    int successfulPurchases = 0;

    scanf("%d", &walletBalance);
    scanf("%d", &N);

    while (i < N) {
        scanf("%d", &purchase);

        if (walletBalance >= purchase) {
            walletBalance -= purchase;
            successfulPurchases++;
        } else {
            break;   // Stop when balance is insufficient
        }

        i++;
    }

    printf("Successful Purchases: %d\n", successfulPurchases);
    printf("Final Balance: %d", walletBalance);

    return 0;
}
