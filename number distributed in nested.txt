#include <stdio.h>

int main() {
    int i, j, n;

    scanf("%d", &n);
    int size = 2 * n-1;
    for(i = 0; i < size; i++) {
       for(j = 0; j < size; j++) {
           int top = i;
           int bottom = size - i - 1;
           int left = j;
           int right = size -j - 1;
           int min = top;
           if(bottom < min)min=bottom;
           if(left < min)min=left;
           if(right < min)min=right;
            printf(" %d",n-min);
        }
      printf("\n");
    }

    return 0;
}
