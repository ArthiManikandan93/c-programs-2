#include <stdio.h>

int main() {
    int N;
    int i = 0;
    int vibration;
    int unsafeCount = 0;
    int consecutiveUnsafe = 0;
    int breakdownReading = -1;

    scanf("%d", &N);

    while (i < N) {
        scanf("%d", &vibration);

        if (vibration > 70) {   // Unsafe reading
            unsafeCount++;
            consecutiveUnsafe++;

            if (consecutiveUnsafe == 3 && breakdownReading == -1) {
                breakdownReading = i + 1;  
            }
        } else {
            consecutiveUnsafe = 0;
        }

        i++;
    }

    if (breakdownReading == -1)
        printf("Breakdown At Reading: Not Occurred\n");
    else
        printf("Breakdown At Reading: %d\n", breakdownReading);

    printf("Unsafe Readings: %d", unsafeCount);

    return 0;
}
