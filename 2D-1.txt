#include<stdio.h>
void main() {

    int R=0;
    int C = 0;
    scanf("%d %d",&R,&C);
    int salary[R][C];
    for (int i=0; i < R;i++) {
        for (int j=0;j<C;j++) {
            scanf("%d",&salary[i][j]);
        }
    }
    int max=0;
    int second=-1;
    for (int i=0;i<R;i++) {
        for (int j=0;j<C;j++) {
            if(salary[i][j]>max) {
                max=salary[i][j];
            }
        }
    }
    for (int i=0;i<R;i++) {
        for (int j=0;j<C;j++) {
            if (salary[i][j]>second && salary[i][j]<max) {
                second=salary[i][j];
            }
        }
    }
    printf("Second salary:%d",second);
    }
