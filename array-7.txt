#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N); 
    int even_count = 0;
    int odd_count = 0;
    int day;

    for(int i = 0; i < N; i++) {
        scanf("%d", &day);
        if(day % 2 == 0)
            even_count++;
        else
            odd_count++;
    }

    printf("Even: %d Odd: %d\n", even_count, odd_count);

    return 0;
}
