#include <stdio.h>

int main() {
    int r, c;
    scanf("%d %d", &r, &c);

    int a[r][c];
    
    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    int count = 0;

    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++) {

            int row_min = a[i][0];
            for(int k = 1; k < c; k++) {
                if(a[i][k] < row_min)
                    row_min = a[i][k];
            }

            int col_max = a[0][j];
            for(int k = 1; k < r; k++) {
                if(a[k][j] > col_max)
                    col_max = a[k][j];
            }

            if(a[i][j] == row_min && a[i][j] == col_max) {
                count++;
            }
        }
    }

    printf("%d", count);

    return 0;
}
