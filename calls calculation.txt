#include <stdio.h>

int main() {
    int N;
    
    scanf("%d", &N);

    int call;
    int totalcalls = 0;
    int overloadhour = 0;

    for(int i = 0; i < N; i++) {
        scanf("%d", &call);
        
        totalcalls = totalcalls + call;

        if(call > 40) {
            overloadhour++;
        }
    }

    printf("total calls: %d\n",  totalcalls);
    printf("overloaed hour: %d\n",overloadhour);

    return 0;
}
