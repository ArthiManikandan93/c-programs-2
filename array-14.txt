#include <stdio.h>
#include<stdio.h>
int main() {
    int n;
    scanf("%d", &n);

    int price[n];
    
    for(int i = 0; i < n; i++) {
        scanf("%d", &price[i]);
    }

    int min = 0;      
    int found = 0;   

    for(int i = 0; i < n; i++) {
        if(price[i] > 0) {
            if(found == 0) {
                min = price[i];
                found = 1;
            }
            else if(price[i] < min) {
                min = price[i];
            }
        }
    }

    if(found == 1)
        printf("%d", min);
    else
        printf("No positive");

    return 0;
}
